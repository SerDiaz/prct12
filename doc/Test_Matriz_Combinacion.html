# File test/tc_matriz_combinacion.rb, line 59
def test_enteros
    #Suma
    @m3[0, 0] = 1
    @m3[0, 1] = 2
    @m3[1, 0] = 5
    @m3[1, 1] = 4
    assert_equal(@m3, @m1+@m2)

    #Resta
    @m3[0, 0] = 1
    @m3[0, 1] = 2
    @m3[1, 0] = 1
    @m3[1, 1] = 4
    assert_equal(@m3, @m1-@m2)

    #Multiplicación
    @m3[0, 0] = 0
    @m3[0, 1] = 0
    @m3[1, 0] = 6
    @m3[1, 1] = 0
    assert_equal(@m3, @m1*@m2)

    #Comparación
    @m3[0, 0] = 0
    @m3[0, 1] = 0
    @m3[1, 0] = 2
    @m3[1, 1] = 0
    assert_equal(@m3, @m2)
end

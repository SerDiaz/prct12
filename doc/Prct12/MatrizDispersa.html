def []=(fila, columna, valor)
  if valor != zero
    @container[Posicion.new(fila, columna)] = valor
  else
    @container.delete(Posicion.new(fila, columna))
  end

# File lib/prct12/matriz_dispersa.rb, line 53
def columnas=(value)
  validate_sizes(value)

  if value < columnas
    @container.each_key do |k|
      if k.c >= value
        @container.delete k
      end
    end
  end

  columnas = value
end
